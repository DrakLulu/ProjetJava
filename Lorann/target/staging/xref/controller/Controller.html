31  	@Override
32  	public void play() 
33  	{
34  		setUserOrder(UserOrder.NOMOVE);
35  		
36  		while (model.getLorann().isAlive() != false) {
37  			
38  			setLorannY(model.getLorann().getY());
39  			setLorannX(model.getLorann().getX());
40  			setTbl(model.getTable());
41  			setDoorX(model.getxDoor());
42  			setDoorY(model.getyDoor());
43  			
44  			movement();
45  			IA();
46  			
47  			
48  			
49  			
50  			char className = tbl[lorannX][lorannY].getSprite();
51  			if (className == Purse.Purse || className == Crystal.Crystal ) 
52  			{
53  				switch(className) 
54  				{
55  					case '1': 
56  						tbl[lorannX][lorannY] = new Nothing();
57  						setScore(score+100);
58  						break; 
59  				
60  					
61  					case 'Q':
62  						tbl[lorannX][lorannY] = new Nothing();
63  						tbl[DoorY][DoorX] = new Open_Door();
64  						this.model.updatedModel();
65  						setScore(score+50);
66  						break; 
67  					
68  					default: 
69  						break; 
70  				} 
71  			}
72  			if (model.isSpellexist() ==true ) {
73  				moveSpell();
74  				this.model.updatedModel();
75  			}
76  		
77  			ifWin(); 
78  			ifDead();
79  			
80  			try {
81  				Thread.sleep(100);
82  			} catch (InterruptedException e) {
83  				e.printStackTrace();
84  			}
85  			this.model.updatedModel();
86  		}
87  	}

133 			int x = m.getY();
134 	        int nombre = (int)(Math.random()*10)%5;
135 			switch (nombre)
136 	        {
137 	            case 1 :
138 	                if (tbl[y+1][x].getPermeability() != Permeability.Blocking)
139 	                {

141 	                }
142 	                else

144 	                m.noMove();
145 	                }
146 	                break;
147

150 	                {
151 	                m.moveUp();
152 	                }
153 	                else 
154 	                {
155 	                m.noMove();
156 	                }
157 	                break;
158 
159 	            case 3 :
160 	                if (tbl[y-1][x].getPermeability() != Permeability.Blocking)
161 	                {
162 	                m.moveLeft();
163 	                }
164 	                else 
165 	                {
166 	                m.noMove();
167 	                }
168 	                break;
169 
170 	            case 4 :

172 	                {
173 	                m.moveDown();
174 	                }
175 	                else 
176 	                {
177 	                m.noMove();
178 	                }
179 	                break;
180

182 	                m.noMove();
183

206 				model.getLorann().loadImage("lorann_b.png");
207 			}
208 			else model.getLorann().noMove();
209 			break;
210 			
211 		case LEFT:

216 				}
217 			model.getLorann().noMove();
218 			break;
219 			
220 		case UP:

223 				model.getLorann().loadImage("lorann_r.png");
224 				}
225 			else model.getLorann().noMove();
226 			break;
227 			
228 		case DOWN:

230 				model.getLorann().moveDown();
231 				model.getLorann().loadImage("lorann_l.png");
232 				
233 				}
234 			else model.getLorann().noMove();
235 			break;

239 			break;
240 			
241 		default:
